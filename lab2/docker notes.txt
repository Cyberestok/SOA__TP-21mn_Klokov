	КОМАНДЫ ДЛЯ ДОКЕРА

// проверка работоспособности докера + перечень команд
docker

// проверка успешности установки докера, проверив статус службы
sudo systemctl status docker

// создание контейнера (запуск образа)
docker run -d -p <номер порта 1>:<номер порта 2> <имя/ID образа>
// "-d" - запустить проект в фоновом режиме
// "номер порта 1" - порт на локальной машине
// "номер порта 2" - порт внутри контейнера

// информация о команде
docker <интересующая команда> --help

// полная инфа по докеру (запущеные, остановленые контейнеры, образы...)
docker info

// инфа по образам
docker images

// список активных (запущеных) контейнеров
docker ps

// список всех доступных контейнеров (активных и неактивных)
docker ps -a

// остановка контейнера
docker stop <имя/ID контейнера>

// запуск контейнера
docker start <имя/ID контейнера>

// поставить контейнер на паузу
docker pause <имя/ID контейнера>

// убрать контейнер из паузы
docker unpause <имя/ID контейнера>

// перезапуск контейнера
docker restart <имя/ID контейнера>

// принудительно завершить работу контейнера
docker kill <имя/ID контейнера>

// авторизация в докер-хаб через терминал (например, чтобы выкладывать свои образы на сайте докер-хаб)
docker login

// выход из учётной записи докера
docker logout

// установка образа последней версии из докер-хаба
docker pull <образ из докер-хаба>
// если нужен образ определённой версии
docker pull <образ из докер-хаба>:<версия>

// запуск образа (если образ не найден, он автоматически сначала скачивается)
// если точнее, то тут создаётся и запускается контейнер, а не образ
docker run <образ>

// запуск образа в интерактивном режиме
docker run -it <образ>

// запуск контейнера + его явная инициализация (задание имени)
docker run -it --name <имя контейнера> <образ>

// "Ctrl + D" - чтобы выйти из контейнера после его запуска

// построить проект (образ)
docker build <путь к докер-файлу>

// удаление образа
docker image rm <ID образа>
// или
docker rmi <имя/ID образа>

// вход внутрь контейнера для внесения изменений
docker exec -it <имя/ID контейнера> bash
